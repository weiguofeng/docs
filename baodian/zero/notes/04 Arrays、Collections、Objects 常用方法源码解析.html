04 ArraysгҖҒCollectionsгҖҒObjects еёёз”Ёж–№жі•жәҗз Ғи§Јжһҗ

й”Ӣж јblog 2020-12-18 Java

иҜ»дёҖжң¬еҘҪд№ҰпјҢе°ұжҳҜе’Ңи®ёеӨҡй«ҳе°ҡзҡ„дәәи°ҲиҜқгҖӮ вҖ”вҖ”жӯҢеҫ·

# еј•еҜјиҜӯ

жҲ‘д»¬еңЁе·ҘдҪңдёӯйғҪдјҡеҶҷе·Ҙе…·зұ»пјҢдҪҶеҰӮдҪ•жүҚиғҪдҪҝеҶҷеҮәжқҘзҡ„е·Ҙе…·зұ»жӣҙеҘҪз”ЁпјҢд№ҹжҳҜжңүдёҖдәӣжҠҖе·§зҡ„гҖӮжң¬з« еҶ…е®№д»Ҙдёүз§Қе№іж—¶е·ҘдҪңдёӯз»ҸеёёдҪҝз”Ёзҡ„е·Ҙе…·зұ»дёәдҫӢпјҢд»ҺдҪҝз”ЁжЎҲдҫӢеҮәеҸ‘пјҢеҶҚзңӢзңӢеә•еұӮжәҗз Ғзҡ„е®һзҺ°пјҢзңӢзңӢиғҪеҗҰеӯҰд№ еҲ°дёҖдәӣе·Ҙе…·зұ»зҡ„жҠҖе·§пјҢд»ҘеҸҠдёүз§Қе·Ҙе…·зұ»зҡ„е®һйҷ…дҪҝз”ЁеңәжҷҜгҖӮ

дёӢж–№жҳҜжң¬дё“ж Ҹ GitHub ең°еқҖпјҡ жәҗз Ғи§Јжһҗпјҡhttps://github.com/luanqiu/java8 (opens new window) ж–Үз«  demoпјҡhttps://github.com/luanqiu/java8_demo (opens new window) еҗҢеӯҰд»¬жңүйңҖиҰҒеҸҜд»ҘеҜ№з…§зқҖжқҘзңӢ пјҡпјү

# 1 е·Ҙе…·зұ»йҖҡз”Ёзҡ„зү№еҫҒ

еҶҚзңӢз»ҶиҠӮд№ӢеүҚпјҢжҲ‘д»¬е…ҲжҖ»з»“дёҖдёӢеҘҪзҡ„е·Ҙе…·зұ»йғҪжңүе“ӘдәӣйҖҡз”Ёзҡ„зү№еҫҒеҶҷжі•пјҡ

  1. жһ„йҖ еҷЁеҝ…йЎ»жҳҜз§Ғжңүзҡ„гҖӮиҝҷж ·зҡ„иҜқпјҢе·Ҙе…·зұ»е°ұж— жі•иў« new еҮәжқҘпјҢеӣ дёәе·Ҙе…·зұ»еңЁдҪҝз”Ёзҡ„ж—¶еҖҷпјҢж— йңҖеҲқе§ӢеҢ–пјҢзӣҙжҺҘдҪҝз”ЁеҚіеҸҜпјҢжүҖд»ҘдёҚдјҡејҖж”ҫеҮәжһ„йҖ еҷЁеҮәжқҘгҖӮ
  2. е·Ҙе…·зұ»зҡ„е·Ҙе…·ж–№жі•еҝ…йЎ»иў« staticгҖҒfinal е…ій”®еӯ—дҝ®йҘ°гҖӮиҝҷж ·зҡ„иҜқе°ұеҸҜд»ҘдҝқиҜҒж–№жі•дёҚеҸҜеҸҳпјҢе№¶дё”еҸҜд»ҘзӣҙжҺҘдҪҝз”ЁпјҢйқһеёёж–№дҫҝгҖӮ

жҲ‘д»¬йңҖиҰҒжіЁж„Ҹзҡ„жҳҜпјҢе°ҪйҮҸдёҚеңЁе·Ҙе…·ж–№жі•дёӯпјҢеҜ№е…ұдә«еҸҳйҮҸжңүеҒҡдҝ®ж”№зҡ„ж“ҚдҪңи®ҝй—®пјҲеҰӮжһңеҝ…йЎ»иҰҒеҒҡзҡ„иҜқпјҢеҝ…йЎ»еҠ й”ҒпјүпјҢеӣ дёәдјҡжңүзәҝзЁӢе®үе…Ёзҡ„й—®йўҳгҖӮйҷӨжӯӨд№ӢеӨ–пјҢе·Ҙе…·зұ»ж–№жі•жң¬иә«жҳҜжІЎжңүзәҝзЁӢе®үе…Ёй—®йўҳзҡ„пјҢеҸҜд»Ҙж”ҫеҝғдҪҝз”ЁгҖӮ

# 2 Arrays

Arrays дё»иҰҒеҜ№ж•°з»„жҸҗдҫӣдәҶдёҖдәӣй«ҳж•Ҳзҡ„ж“ҚдҪңпјҢжҜ”еҰӮиҜҙжҺ’еәҸгҖҒжҹҘжүҫгҖҒеЎ«е……гҖҒжӢ·иҙқгҖҒзӣёзӯүеҲӨж–ӯзӯүзӯүгҖӮжҲ‘д»¬йҖүжӢ©е…¶дёӯдёӨдёүзңӢдёӢпјҢеҜ№е…¶дҪҷж“ҚдҪңж„ҹе…ҙи¶Јзҡ„еҗҢеӯҰеҸҜд»ҘеҲ° GitHub дёҠжҹҘзңӢжәҗз Ғи§ЈжһҗгҖӮ

# 2.1 жҺ’еәҸ

Arrays.sort ж–№жі•дё»иҰҒз”ЁдәҺжҺ’еәҸпјҢе…ҘеҸӮж”ҜжҢҒ intгҖҒlongгҖҒdouble зӯүеҗ„з§Қеҹәжң¬зұ»еһӢзҡ„ж•°з»„пјҢд№ҹж”ҜжҢҒиҮӘе®ҡд№үзұ»зҡ„ж•°з»„пјҢдёӢйқўжҲ‘д»¬еҶҷдёӘ demo жқҘжј”зӨәдёҖдёӢиҮӘе®ҡд№үзұ»ж•°з»„зҡ„жҺ’еәҸпјҡ

@Data
// иҮӘе®ҡд№үзұ»
class SortDTO {
  private String sortTarget;
  public SortDTO(String sortTarget) {
    this.sortTarget = sortTarget;
  }
}
@Test
public void testSort(){
  List<SortDTO> list = ImmutableList.of(
      new SortDTO("300"),
      new SortDTO("50"),
      new SortDTO("200"),
      new SortDTO("220")
  );
  // жҲ‘д»¬е…ҲжҠҠж•°з»„зҡ„еӨ§е°ҸеҲқе§ӢеҢ–жҲҗ list зҡ„еӨ§е°ҸпјҢдҝқиҜҒиғҪеӨҹжӯЈзЎ®жү§иЎҢ toArray
  SortDTO[] array = new SortDTO[list.size()];
  list.toArray(array);
  log.info("жҺ’еәҸд№ӢеүҚпјҡ{}", JSON.toJSONString(array));
  Arrays.sort(array, Comparator.comparing(SortDTO::getSortTarget));
  log.info("жҺ’еәҸд№ӢеҗҺпјҡ{}", JSON.toJSONString(array));
}

д»Һиҫ“еҮәзҡ„з»“жһңдёӯеҸҜд»ҘзңӢеҲ°пјҢжҺ’еәҸд№ӢеҗҺзҡ„ж•°з»„е·Із»ҸжҳҜжңүйЎәеәҸзҡ„дәҶпјҢд№ҹеҸҜд»ҘзңӢеҲ° sort ж–№жі•ж”ҜжҢҒдёӨдёӘе…ҘеҸӮпјҡиҰҒжҺ’еәҸзҡ„ж•°з»„е’ҢеӨ–йғЁжҺ’еәҸеҷЁгҖӮ еӨ§е®¶йғҪиҜҙ sort ж–№жі•жҺ’еәҸзҡ„жҖ§иғҪиҫғй«ҳпјҢдё»иҰҒеҺҹеӣ жҳҜ sort дҪҝз”ЁдәҶеҸҢиҪҙеҝ«йҖҹжҺ’еәҸз®—жі•пјҢе…·дҪ“з®—жі•е°ұдёҚз»ҶиҜҙдәҶгҖӮ

# 2.1 дәҢеҲҶжҹҘжүҫжі•

Arrays.binarySearch ж–№жі•дё»иҰҒз”ЁдәҺеҝ«йҖҹд»Һж•°з»„дёӯжҹҘжүҫеҮәеҜ№еә”зҡ„еҖјгҖӮе…¶ж”ҜжҢҒзҡ„е…ҘеҸӮзұ»еһӢйқһеёёеӨҡпјҢеҰӮ byteгҖҒintгҖҒlong еҗ„з§Қзұ»еһӢзҡ„ж•°з»„гҖӮиҝ”еӣһеҸӮж•°жҳҜжҹҘжүҫеҲ°зҡ„еҜ№еә”ж•°з»„дёӢж Үзҡ„еҖјпјҢеҰӮжһңжҹҘиҜўдёҚеҲ°пјҢеҲҷиҝ”еӣһиҙҹж•°гҖӮ жҲ‘д»¬еҶҷдәҶдёҖдёӘ demo еҰӮдёӢпјҡ

List<SortDTO> list = ImmutableList.of(
    new SortDTO("300"),
    new SortDTO("50"),
    new SortDTO("200"),
    new SortDTO("220")
);
SortDTO[] array = new SortDTO[list.size()];

д»ҺдёҠиҝ°д»Јз ҒдёӯжҲ‘д»¬йңҖиҰҒжіЁж„ҸдёӨзӮ№пјҡ

  1. еҰӮжһңиў«жҗңзҙўзҡ„ж•°з»„жҳҜж— еәҸзҡ„пјҢдёҖе®ҡиҰҒе…ҲжҺ’еәҸпјҢеҗҰеҲҷдәҢеҲҶжҗңзҙўеҫҲжңүеҸҜиғҪжҗңзҙўдёҚеҲ°пјҢжҲ‘д»¬ demo йҮҢйқўд№ҹе…ҲеҜ№ж•°з»„иҝӣиЎҢдәҶжҺ’еәҸпјӣ
  2. жҗңзҙўж–№жі•иҝ”еӣһзҡ„жҳҜж•°з»„зҡ„дёӢж ҮеҖјгҖӮеҰӮжһңжҗңзҙўдёҚеҲ°пјҢиҝ”еӣһзҡ„дёӢж ҮеҖје°ұдјҡжҳҜиҙҹж•°пјҢиҝҷж—¶жҲ‘д»¬йңҖиҰҒеҲӨж–ӯдёҖдёӢжӯЈиҙҹгҖӮеҰӮжһңжҳҜиҙҹж•°пјҢиҝҳд»Һж•°з»„дёӯиҺ·еҸ–ж•°жҚ®зҡ„иҜқпјҢдјҡжҠҘж•°з»„и¶Ҡз•Ңзҡ„й”ҷиҜҜгҖӮdemo дёӯеҜ№иҝҷз§Қжғ…еҶөиҝӣиЎҢдәҶеҲӨж–ӯпјҢеҰӮжһңжҳҜиҙҹж•°пјҢдјҡжҸҗеүҚжҠӣеҮәжҳҺзЎ®зҡ„ејӮеёёгҖӮ

жҺҘдёӢжқҘпјҢжҲ‘д»¬жқҘзңӢдёӢдәҢеҲҶжі•еә•еұӮд»Јз Ғзҡ„е®һзҺ°пјҡ

// aпјҡжҲ‘д»¬иҰҒжҗңзҙўзҡ„ж•°з»„пјҢfromIndexпјҡд»ҺйӮЈйҮҢејҖе§ӢжҗңзҙўпјҢй»ҳи®ӨжҳҜ0пјӣ toIndexпјҡжҗңзҙўеҲ°дҪ•ж—¶еҒңжӯўпјҢй»ҳи®ӨжҳҜж•°з»„еӨ§е°Ҹ
// keyпјҡжҲ‘д»¬йңҖиҰҒжҗңзҙўзҡ„еҖј cпјҡеӨ–йғЁжҜ”иҫғеҷЁ
private static <T> int binarySearch0(T[] a, int fromIndex, int toIndex,
                                     T key, Comparator<? super T> c) {
    // еҰӮжһңжҜ”иҫғеҷЁ c жҳҜз©әзҡ„пјҢзӣҙжҺҘдҪҝз”Ё key зҡ„ Comparable.compareTo ж–№жі•иҝӣиЎҢжҺ’еәҸ
    // еҒҮи®ҫ key зұ»еһӢжҳҜ String зұ»еһӢпјҢString й»ҳи®Өе®һзҺ°дәҶ Comparable жҺҘеҸЈпјҢе°ұеҸҜд»ҘзӣҙжҺҘдҪҝз”Ё compareTo ж–№жі•иҝӣиЎҢжҺ’еәҸ
    if (c == null) {
        // иҝҷжҳҜеҸҰеӨ–дёҖдёӘж–№жі•пјҢдҪҝз”ЁеҶ…йғЁжҺ’еәҸеҷЁиҝӣиЎҢжҜ”иҫғзҡ„ж–№жі•
        return binarySearch0(a, fromIndex, toIndex, key);
    }
    int low = fromIndex;
    int high = toIndex - 1;
    // ејҖе§ӢдҪҚзҪ®е°ҸдәҺз»“жқҹдҪҚзҪ®пјҢе°ұдјҡдёҖзӣҙеҫӘзҺҜжҗңзҙў
    while (low <= high) {
        // еҒҮи®ҫ low =0пјҢhigh =10пјҢйӮЈд№Ҳ mid е°ұжҳҜ 5пјҢжүҖд»ҘиҜҙдәҢеҲҶзҡ„ж„ҸжҖқдё»иҰҒеңЁиҝҷйҮҢпјҢжҜҸж¬ЎйғҪжҳҜи®Ўз®—зҙўеј•зҡ„дёӯй—ҙеҖј
        int mid = (low + high) >>> 1;
        T midVal = a[mid];
        // жҜ”иҫғж•°з»„дёӯй—ҙеҖје’Ңз»ҷе®ҡзҡ„еҖјзҡ„еӨ§е°Ҹе…ізі»
        int cmp = c.compare(midVal, key);
        // еҰӮжһңж•°з»„дёӯй—ҙеҖје°ҸдәҺз»ҷе®ҡзҡ„еҖјпјҢиҜҙжҳҺжҲ‘д»¬иҰҒжүҫзҡ„еҖјеңЁдёӯй—ҙеҖјзҡ„еҸіиҫ№
        if (cmp < 0)
            low = mid + 1;
        // жҲ‘д»¬иҰҒжүҫзҡ„еҖјеңЁдёӯй—ҙеҖјзҡ„е·Ұиҫ№
        else if (cmp > 0)
            high = mid - 1;
        else
        // жүҫеҲ°дәҶ
            return mid; // key found
    }
    // иҝ”еӣһзҡ„еҖјжҳҜиҙҹж•°пјҢиЎЁзӨәжІЎжңүжүҫеҲ°
    return -(low + 1);  // key not found.
}

# 2.2 жӢ·иҙқ

ж•°з»„жӢ·иҙқжҲ‘д»¬з»ҸеёёйҒҮеҲ°пјҢжңүж—¶йңҖиҰҒжӢ·иҙқж•ҙдёӘж•°з»„пјҢжңүж—¶йңҖиҰҒжӢ·иҙқйғЁеҲҶпјҢжҜ”еҰӮ ArrayList еңЁ addпјҲжү©е®№пјү жҲ– removeпјҲеҲ йҷӨе…ғзҙ дёҚжҳҜжңҖеҗҺдёҖдёӘпјү ж“ҚдҪңж—¶пјҢдјҡиҝӣиЎҢдёҖдәӣжӢ·иҙқгҖӮжӢ·иҙқж•ҙдёӘж•°з»„жҲ‘д»¬еҸҜд»ҘдҪҝз”Ё copyOf ж–№жі•пјҢжӢ·иҙқйғЁеҲҶжҲ‘д»¬еҸҜд»ҘдҪҝз”Ё copyOfRange ж–№жі•пјҢд»Ҙ copyOfRange дёәдҫӢпјҢзңӢдёӢеә•еұӮжәҗз Ғзҡ„е®һзҺ°пјҡ

// original еҺҹе§Ӣж•°з»„ж•°жҚ®
// from жӢ·иҙқиө·зӮ№
// to жӢ·иҙқз»ҲзӮ№
public static char[] copyOfRange(char[] original, int from, int to) {
    // йңҖиҰҒжӢ·иҙқзҡ„й•ҝеәҰ
    int newLength = to - from;
    if (newLength < 0)
        throw new IllegalArgumentException(from + " > " + to);
    // еҲқе§ӢеҢ–ж–°ж•°з»„
    char[] copy = new char[newLength];
    // и°ғз”Ё native ж–№жі•иҝӣиЎҢжӢ·иҙқпјҢеҸӮж•°зҡ„ж„ҸжҖқеҲҶеҲ«жҳҜпјҡ
    // иў«жӢ·иҙқзҡ„ж•°з»„гҖҒд»Һж•°з»„йӮЈйҮҢејҖе§ӢгҖҒзӣ®ж Үж•°з»„гҖҒд»Һзӣ®зҡ„ж•°з»„йӮЈйҮҢејҖе§ӢжӢ·иҙқгҖҒжӢ·иҙқзҡ„й•ҝеәҰ
    System.arraycopy(original, from, copy, 0,
                     Math.min(original.length - from, newLength));
    return copy;
}

# 3 Collections

Collections жҳҜдёәдәҶж–№дҫҝдҪҝз”ЁйӣҶеҗҲиҖҢдә§з”ҹзҡ„е·Ҙе…·зұ»пјҢArrays ж–№дҫҝж•°з»„дҪҝз”ЁпјҢCollections жҳҜж–№дҫҝйӣҶеҗҲдҪҝз”ЁгҖӮ Collections д№ҹжҸҗдҫӣдәҶ sort е’Ң binarySearch ж–№жі•пјҢsort еә•еұӮдҪҝз”Ёзҡ„е°ұжҳҜ Arrays.sort ж–№жі•пјҢbinarySearch еә•еұӮжҳҜиҮӘе·ұйҮҚеҶҷдәҶдәҢеҲҶжҹҘжүҫз®—жі•пјҢе®һзҺ°зҡ„йҖ»иҫ‘е’Ң Arrays зҡ„дәҢеҲҶжҹҘжүҫз®—жі•е®Ңе…ЁдёҖиҮҙпјҢиҝҷдёӨдёӘж–№жі•дёҠ Collections е’Ң Arrays зҡ„еҶ…йғЁе®һзҺ°еҫҲзұ»дјјпјҢжҺҘдёӢжқҘжҲ‘д»¬жқҘзңӢдёӢ Collections зӢ¬жңүзҡ„зү№жҖ§гҖӮ

# 3.1 жұӮйӣҶеҗҲдёӯжңҖеӨ§гҖҒе°ҸеҖј

жҸҗдҫӣдәҶ max ж–№жі•жқҘеҸ–еҫ—йӣҶеҗҲдёӯзҡ„жңҖеӨ§еҖјпјҢmin ж–№жі•жқҘеҸ–еҫ—йӣҶеҗҲдёӯзҡ„жңҖе°ҸеҖјпјҢmax е’Ң min ж–№жі•еҫҲзӣёдјјзҡ„пјҢжҲ‘д»¬д»Ҙ max ж–№жі•дёәдҫӢжқҘиҜҙжҳҺдёҖдёӢпјҢmax жҸҗдҫӣдәҶдёӨз§Қзұ»еһӢзҡ„ж–№жі•пјҢдёҖдёӘйңҖиҰҒдј еӨ–йғЁжҺ’еәҸеҷЁпјҢдёҖдёӘдёҚйңҖиҰҒдј жҺ’еәҸеҷЁпјҢдҪҶйңҖиҰҒйӣҶеҗҲдёӯзҡ„е…ғзҙ ејәеҲ¶е®һзҺ° Comparable жҺҘеҸЈпјҢеҗҺиҖ…зҡ„жіӣеһӢе®ҡд№үеҫҲжңүж„ҸжҖқпјҢжҲ‘д»¬жқҘзңӢдёӢпјҲд»ҺеҸіеҫҖе·ҰзңӢпјүпјҡ д»Һиҝҷж®өжәҗз ҒдёӯпјҢжҲ‘д»¬еҸҜд»ҘеӯҰд№ еҲ°дёӨзӮ№пјҡ

  1. max ж–№жі•жіӣеһӢ T е®ҡд№үеҫ—йқһеёёе·§еҰҷпјҢж„ҸжҖқжҳҜжіӣеһӢеҝ…йЎ»з»§жүҝ Object е№¶дё”е®һзҺ° Comparable зҡ„жҺҘеҸЈгҖӮдёҖиҲ¬и®©жҲ‘д»¬жқҘе®ҡд№үзҡ„иҜқпјҢжҲ‘д»¬еҸҜд»ҘдјҡеңЁж–№жі•йҮҢйқўеҺ»еҲӨж–ӯжңүж— е®һзҺ° Comparable зҡ„жҺҘеҸЈпјҢиҝҷз§ҚжҳҜеңЁиҝҗиЎҢж—¶жүҚиғҪзҹҘйҒ“з»“жһңгҖӮиҖҢиҝҷйҮҢжіӣеһӢзӣҙжҺҘе®ҡд№үдәҶеҝ…йЎ»е®һзҺ° Comparable жҺҘеҸЈпјҢеңЁзј–иҜ‘зҡ„ж—¶еҖҷе°ұеҸҜе‘ҠиҜүдҪҝз”ЁиҖ…пјҢеҪ“еүҚзұ»жІЎжңүе®һзҺ° Comparable жҺҘеҸЈпјҢдҪҝз”Ёиө·жқҘеҫҲеҸӢеҘҪпјӣ
  2. з»ҷжҲ‘д»¬жҸҗдҫӣдәҶе®һзҺ°дёӨз§ҚжҺ’еәҸжңәеҲ¶зҡ„еҘҪзӨәдҫӢпјҡиҮӘе®ҡд№үзұ»е®һзҺ° Comparable жҺҘеҸЈе’Ңдј е…ҘеӨ–йғЁжҺ’еәҸеҷЁгҖӮдёӨз§ҚжҺ’еәҸе®һзҺ°еҺҹзҗҶзұ»дјјпјҢдҪҶе®һзҺ°жңүжүҖе·®еҲ«пјҢжҲ‘д»¬еңЁе·ҘдҪңдёӯеҰӮжһңйңҖиҰҒдәӣжҺ’еәҸзҡ„е·Ҙе…·зұ»ж—¶пјҢеҸҜд»Ҙж•Ҳд»ҝгҖӮ

# 3.2 еӨҡз§Қзұ»еһӢзҡ„йӣҶеҗҲ

Collections еҜ№еҺҹе§ӢйӣҶеҗҲзұ»иҝӣиЎҢдәҶе°ҒиЈ…пјҢжҸҗдҫӣдәҶжӣҙеҘҪзҡ„йӣҶеҗҲзұ»з»ҷжҲ‘д»¬пјҢдёҖз§ҚжҳҜзәҝзЁӢе®үе…Ёзҡ„йӣҶеҗҲпјҢдёҖз§ҚжҳҜдёҚеҸҜеҸҳзҡ„йӣҶеҗҲпјҢй’ҲеҜ№ ListгҖҒMapгҖҒSet йғҪжңүжҸҗдҫӣпјҢжҲ‘д»¬е…ҲжқҘзңӢдёӢзәҝзЁӢе®үе…Ёзҡ„йӣҶеҗҲпјҡ

# 3.2.1 зәҝзЁӢе®үе…Ёзҡ„йӣҶеҗҲ

зәҝзЁӢе®үе…Ёзҡ„йӣҶеҗҲж–№жі•йғҪжҳҜ synchronized жү“еӨҙзҡ„пјҢеҰӮдёӢпјҡ д»Һж–№жі•е‘ҪеҗҚжҲ‘д»¬йғҪеҸҜд»ҘзңӢеҮәжқҘпјҢеә•еұӮжҳҜйҖҡиҝҮ synchronized иҪ»йҮҸй”ҒжқҘе®һзҺ°зҡ„пјҢжҲ‘д»¬д»Ҙ synchronizedList дёәдҫӢжқҘиҜҙжҳҺдёӢеә•еұӮзҡ„е®һзҺ°пјҡ еҸҜд»ҘзңӢеҲ° List зҡ„жүҖжңүж“ҚдҪңж–№жі•йғҪиў«еҠ дёҠдәҶ synchronized й”ҒпјҢжүҖд»ҘеӨҡзәҝзЁӢеҜ№йӣҶеҗҲеҗҢж—¶иҝӣиЎҢж“ҚдҪңпјҢжҳҜзәҝзЁӢе®үе…Ёзҡ„гҖӮ

# 3.2.1 дёҚеҸҜеҸҳзҡ„йӣҶеҗҲ

еҫ—еҲ°дёҚеҸҜеҸҳйӣҶеҗҲзҡ„ж–№жі•йғҪжҳҜд»Ҙ unmodifiable ејҖеӨҙзҡ„гҖӮиҝҷзұ»ж–№жі•зҡ„ж„ҸжҖқжҳҜпјҢжҲ‘д»¬дјҡд»ҺеҺҹйӣҶеҗҲдёӯпјҢеҫ—еҲ°дёҖдёӘдёҚеҸҜеҸҳзҡ„ж–°йӣҶеҗҲпјҢж–°йӣҶеҗҲеҸӘиғҪи®ҝй—®пјҢж— жі•дҝ®ж”№пјӣдёҖж—Ұдҝ®ж”№пјҢе°ұдјҡжҠӣеҮәејӮеёёгҖӮиҝҷдё»иҰҒжҳҜеӣ дёәеҸӘејҖж”ҫдәҶжҹҘиҜўж–№жі•пјҢе…¶дҪҷд»»дҪ•дҝ®ж”№ж“ҚдҪңйғҪдјҡжҠӣеҮәејӮеёёпјҢжҲ‘д»¬д»Ҙ unmodifiableList дёәдҫӢжқҘзңӢдёӢеә•еұӮе®һзҺ°жңәеҲ¶пјҡ

# 3.2.2 е°Ҹз»“

д»ҘдёҠдёӨз§Қ List е…¶е®һи§ЈеҶідәҶе·ҘдҪңдёӯзҡ„дёҖдәӣеӣ°жғ‘пјҢжҜ”еҰӮиҜҙ ArrayList жҳҜзәҝзЁӢдёҚе®үе…Ёзҡ„пјҢз„¶еҗҺе…¶еҶ…йғЁж•°з»„еҫҲе®№жҳ“иў«дҝ®ж”№пјҢжңүзҡ„ж—¶еҖҷпјҢжҲ‘д»¬еёҢжңӣ List дёҖж—Ұз”ҹжҲҗеҗҺпјҢе°ұдёҚиғҪиў«дҝ®ж”№пјҢCollections еҜ№ List йҮҚж–°иҝӣиЎҢдәҶе°ҒиЈ…пјҢжҸҗдҫӣдәҶдёӨз§Қзұ»еһӢзҡ„йӣҶеҗҲе°ҒиЈ…еҪўејҸпјҢд»ҺиҖҢи§ЈеҶідәҶе·ҘдҪңдёӯзҡ„дёҖдәӣзғҰжҒјпјҢеҰӮжһңдҪ е№іж—¶дҪҝз”Ё List ж—¶жңүдёҖдәӣзғҰжҒјпјҢд№ҹеҸҜд»ҘеӯҰд№ жӯӨз§Қж–№ејҸпјҢиҮӘе·ұеҜ№еҺҹе§ӢйӣҶеҗҲиҝӣиЎҢе°ҒиЈ…пјҢжқҘи§ЈеҶі List дҪҝз”ЁиҝҮзЁӢдёӯзҡ„дёҚж–№дҫҝгҖӮ

# 4 Objects

еҜ№дәҺ ObjectsпјҢжҲ‘д»¬з»ҸеёёдҪҝз”Ёзҡ„е°ұжҳҜдёӨдёӘеңәжҷҜпјҢзӣёзӯүеҲӨж–ӯе’ҢеҲӨз©әгҖӮ

# 4.1 зӣёзӯүеҲӨж–ӯ

Objects жңүжҸҗдҫӣ equals е’Ң deepEquals дёӨдёӘж–№жі•жқҘиҝӣиЎҢзӣёзӯүеҲӨж–ӯпјҢеүҚиҖ…жҳҜеҲӨж–ӯеҹәжң¬зұ»еһӢе’ҢиҮӘе®ҡд№үзұ»зҡ„пјҢеҗҺиҖ…жҳҜз”ЁжқҘеҲӨж–ӯж•°з»„зҡ„пјҢжҲ‘д»¬жқҘзңӢдёӢеә•еұӮзҡ„жәҗз Ғе®һзҺ°пјҡ д»Һжәҗз ҒдёӯпјҢеҸҜд»ҘзңӢеҮә Objects еҜ№еҹәжң¬зұ»еһӢе’ҢеӨҚжқӮзұ»еһӢзҡ„еҜ№иұЎпјҢйғҪжңүзқҖжҜ”иҫғз»ҶзІ’еәҰзҡ„еҲӨж–ӯпјҢеҸҜд»Ҙж”ҫеҝғдҪҝз”ЁгҖӮ

# 4.2 дёәз©әеҲӨж–ӯ

Objects жҸҗдҫӣдәҶеҗ„з§Қе…ідәҺз©әзҡ„дёҖдәӣеҲӨж–ӯпјҢisNull е’Ң nonNull еҜ№дәҺеҜ№иұЎжҳҜеҗҰдёәз©әиҝ”еӣһ Boolean еҖјпјҢrequireNonNull ж–№жі•жӣҙеҠ дёҘж јпјҢеҰӮжһңдёҖж—Ұдёәз©әпјҢдјҡзӣҙжҺҘжҠӣеҮәејӮеёёпјҢжҲ‘д»¬йңҖиҰҒж №жҚ®з”ҹжҙ»зҡ„еңәжҷҜйҖүжӢ©дҪҝз”ЁгҖӮ

# 5 йқўиҜ•йўҳ

# 5.1 е·ҘдҪңдёӯжңүжІЎжңүйҒҮеҲ°зү№еҲ«еҘҪз”Ёзҡ„е·Ҙе…·зұ»пјҢеҰӮдҪ•еҶҷеҘҪдёҖдёӘе·Ҙе…·зұ»

зӯ”пјҡжңүзҡ„пјҢеғҸ Arrays зҡ„жҺ’еәҸгҖҒдәҢеҲҶжҹҘжүҫгҖҒCollections зҡ„дёҚеҸҜеҸҳгҖҒзәҝзЁӢе®үе…ЁйӣҶеҗҲзұ»гҖҒObjects зҡ„еҲӨз©әзӣёзӯүеҲӨж–ӯзӯүзӯүе·Ҙе…·зұ»пјҢеҘҪзҡ„е·Ҙе…·зұ»иӮҜе®ҡеҫҲеҘҪз”ЁпјҢжҜ”еҰӮиҜҙдҪҝз”Ё static final е…ій”®еӯ—еҜ№ж–№жі•иҝӣиЎҢдҝ®йҘ°пјҢе·Ҙе…·зұ»жһ„йҖ еҷЁеҝ…йЎ»жҳҜз§ҒжңүзӯүзӯүжүӢж®өжқҘеҶҷеҘҪе·Ҙе…·зұ»гҖӮ

# 5.2 еҶҷдёҖдёӘдәҢеҲҶжҹҘжүҫз®—жі•зҡ„е®һзҺ°

зӯ”пјҡеҸҜд»ҘеҸӮиҖғ Arrays зҡ„ binarySearch ж–№жі•зҡ„жәҗз Ғе®һзҺ°гҖӮ

зӯ”пјҡеҸҜд»ҘдҪҝз”Ё Collections зҡ„ unmodifiableList зҡ„ж–№жі•пјҢиҜҘж–№жі•дјҡиҝ”еӣһдёҖдёӘдёҚиғҪиў«дҝ®ж”№зҡ„еҶ…йғЁзұ»йӣҶеҗҲпјҢиҝҷдәӣйӣҶеҗҲзұ»еҸӘејҖж”ҫжҹҘиҜўзҡ„ж–№жі•пјҢеҜ№дәҺи°ғз”Ёдҝ®ж”№йӣҶеҗҲзҡ„ж–№жі•дјҡзӣҙжҺҘжҠӣеҮәејӮеёёгҖӮ
